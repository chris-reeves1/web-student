Level 1 — Simple Guided

What you are building:
A CUSTOMERS array in JS
A function that draws rows into #customersTbody
A function that filters customers based on a search string
An init() that wires everything up

Definition of Done:
Page loads and shows all customers in the table
Typing “france” filters the table
Counter updates correctly every time you type

Step-by-step:

Create: js/app-day3.js and add your script tag in index.html.

Step 1 — Create the data (write it or copy from samle_data_js)

Create an array with at least 7 customers:

const CUSTOMERS = [
  { CustomerID: "ALFKI", CompanyName: "Alfreds Futterkiste", Country: "Germany" },
  // add at least 6 more
];

Step 2 — Get the elements

Create variables for the three elements:
customersTbody
customerSearch
customerCounter

Use document.getElementById(...).

Step 3 — Write renderCustomers(customers) 

Rules:
Clears the table first
Adds 1 <tr> per customer
Each row must have 3 <td> cells: ID, Name, Country

If customers.length === 0, show a single row: “No customers match your search.” spanning 3 columns

Hint:
customersTbody.innerHTML = "";


Step 4 — Write filterCustomers(query) 

Rules:
Returns all customers if query is blank
Otherwise match if query appears in:
CustomerID
CompanyName
Country

Must be case-insensitive (use .toLowerCase())

Step 5 — Write init() 

Rules:
On page load: renderCustomers(CUSTOMERS)
Set the counter: X shown out of Y total
Add an input event listener to the search box:
    Read search.value
    Filter
    Render filtered
    Update counter

Wire it up with:
document.addEventListener("DOMContentLoaded", init);


simple solution:
"use strict";

/* 1) Data (in-memory, no JSON yet) */
const CUSTOMERS = [
  { CustomerID: "ALFKI", CompanyName: "Alfreds Futterkiste", Country: "Germany" },
  { CustomerID: "ANATR", CompanyName: "Ana Trujillo Emparedados", Country: "Mexico" },
  { CustomerID: "AROUT", CompanyName: "Around the Horn", Country: "United Kingdom" },
  { CustomerID: "BONAP", CompanyName: "Bon app'", Country: "France" },
  { CustomerID: "BSBEV", CompanyName: "B's Beverages", Country: "United Kingdom" },
  { CustomerID: "FRANK", CompanyName: "Frankenversand", Country: "Germany" },
  { CustomerID: "VINET", CompanyName: "Vins et alcools Chevalier", Country: "France" },
];

/* 2) Entry point */
function init() {
  const tbody = document.getElementById("customersTbody");
  const search = document.getElementById("customerSearch");
  const counter = document.getElementById("customerCounter");

  // initial render
  renderCustomers(tbody, CUSTOMERS);
  counter.textContent = `${CUSTOMERS.length} shown out of ${CUSTOMERS.length} total`;

  // live search
  search.addEventListener("input", () => {
    const query = search.value.trim().toLowerCase();
    const filtered = filterCustomers(CUSTOMERS, query);

    renderCustomers(tbody, filtered);
    counter.textContent = `${filtered.length} shown out of ${CUSTOMERS.length} total`;
  });
}

/* 3) Render */
function renderCustomers(tbody, customers) {
  tbody.innerHTML = "";

  if (customers.length === 0) {
    tbody.innerHTML = `<tr><td colspan="3">No customers match your search.</td></tr>`;
    return;
  }

  for (const c of customers) {
    tbody.innerHTML += `
      <tr class="row is-clickable">
        <td>${c.CustomerID}</td>
        <td>${c.CompanyName}</td>
        <td>${c.Country}</td>
      </tr>
    `;
  }
}

/* 4) Filter */
function filterCustomers(customers, query) {
  if (!query) return customers;

  return customers.filter((c) => {
    const id = c.CustomerID.toLowerCase();
    const name = c.CompanyName.toLowerCase();
    const country = c.Country.toLowerCase();
    return id.includes(query) || name.includes(query) || country.includes(query);
  });
}

/* Start */
document.addEventListener("DOMContentLoaded", init);
